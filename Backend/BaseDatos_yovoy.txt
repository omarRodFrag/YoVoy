CREATE DATABASE yovoy;
use yovoy

CREATE TABLE rutas (
    idRuta SERIAL PRIMARY KEY,
    nombre VARCHAR(50),
    descripcion TEXT
);

CREATE TABLE unidades (
    idUnidad SERIAL PRIMARY KEY,
    nombre VARCHAR(50),
    numero_economico VARCHAR(10) UNIQUE,
    latitud DECIMAL(10, 8),
    longitud DECIMAL(11, 8),
    ultima_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ruta_id INT REFERENCES rutas(id) ON DELETE SET NULL
);

CREATE TABLE paradas (
    idParadas SERIAL PRIMARY KEY,
    nombre VARCHAR(50),
    latitud DECIMAL(10, 8),
    longitud DECIMAL(11, 8)
);

CREATE TABLE ruta_parada (
    idRuta_Parada SERIAL PRIMARY KEY,
    ruta_id INT REFERENCES rutas(id) ON DELETE CASCADE,
    parada_id INT REFERENCES paradas(id) ON DELETE CASCADE,
    UNIQUE (ruta_id, parada_id)  -- Evita duplicados en la relaci√≥n
);

CREATE TABLE Usuarios (
    idUsuario INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,         
    apellidos VARCHAR(100) NOT NULL,      
    email VARCHAR(100) UNIQUE NOT NULL,   
    password VARCHAR(255) NOT NULL,
    rol ENUM('admin', 'user', 'editor') NOT NULL DEFAULT 'user'
);
